## 4. 어드민 API 추가 (자동생성 콘텐츠 관리용)

```typescript
// src/admin/admin.controller.ts에 추가

// 자동생성 DRAFT 목록 조회
@Get('auto-content')
@UseGuards(AdminGuard)
async getAutoContent(
  @Query('status') status?: string,  // DRAFT | PUBLISHED | ALL
  @Query('page') page?: number,
) {
  return this.postsService.findMany({
    where: {
      isAutoGenerated: true,
      ...(status && status !== 'ALL' ? { status } : {}),
    },
    orderBy: { createdAt: 'desc' },
    take: 20,
    skip: ((page || 1) - 1) * 20,
  });
}

// DRAFT → PUBLISHED 승인
@Patch('auto-content/:id/approve')
@UseGuards(AdminGuard)
async approveContent(@Param('id') id: string) {
  return this.postsService.update(id, { status: 'PUBLISHED' });
}

// DRAFT 삭제 (거부)
@Delete('auto-content/:id')
@UseGuards(AdminGuard)
async rejectContent(@Param('id') id: string) {
  return this.postsService.delete(id);
}
```
